################################################################################
# TEST 1 — два действительных корня (базовый)
################################################################################
./app.out -q 1e-6 1 -3 2   # перестановки (A,B,C):
                           # (1, -3, 2)   → x = 1, 2
                           # (1, 2, -3)   → x = -3, 1
                           # (-3, 1, 2)   → x = 1, -2/3
                           # (-3, 2, 1)   → x = 1, -1/3
                           # (2, 1, -3)   → x = -3/2, 1
                           # (2, -3, 1)   → x = 1/2, 1


################################################################################
# TEST 2 — кратный корень (D == 0)
################################################################################
./app.out -q 1e-6 1 2 1    # перестановки:
                           # (1, 2, 1)   → кратный корень: -1
                           # (1, 1, 2)   → x = -2, -1
                           # (2, 1, 1)   → кратный: -1/2
                           # остальные 3 перестановки — дубликаты (коэффициенты повторяются)


################################################################################
# TEST 3 — нет действительных корней (D < 0)
################################################################################
./app.out -q 1e-6 1 0 1    # перестановки:
                           # (1, 0, 1)  → нет действительных корней
                           # (1, 1, 0)  → x = 0, -1
                           # (0, 1, 1)  → линейное: x = -1
                           # остальные 3 перестановки — дубликаты


################################################################################
# TEST 4 — вырождение (A ≈ 0)
################################################################################
./app.out -q 1e-6 0 2 -4  # перестановки:
                          # (0, 2, -4) → линейное: x = 2
                          # (0, -4, 2) → линейное: x = -0.5
                          # (2, 0, -4) → x = ±1.41421
                          # (2, -4, 0) → x = 0, 2
                          # (-4, 0, 2) → x = ±0.70711
                          # (-4, 2, 0) → x = 0, -0.5


################################################################################
# TEST 5 — полное вырождение 0x² + 0x + C = 0
################################################################################
./app.out -q 1e-6 0 0 5   # перестановки:
                          # (0,0,5) → нет решений
                          # (0,5,0) → x = 0
                          # (5,0,0) → кратный корень x = 0
                          # остальные повторяются


################################################################################
# TEST 6 — тождество 0 = 0 (бесконечно много решений)
################################################################################
./app.out -q 1e-6 0 0 0   # все 6 перестановок:
                          # тождество → бесконечно много решений


# --- Кратность целых (-m a b) ---
./app.out -m 10 5                  # кратно
./app.out -m 10 3                  # не кратно
./app.out -m -12 3                 # отрицательное первое, кратно
./app.out -m 12 -3                 # отрицательное второе, кратно
./app.out -m -12 -3                # оба отрицательные, кратно
./app.out -m 7 1                   # кратно единице
./app.out -m 1 7                   # не кратно
./app.out -m 2147483646 2          # большие числа (перед переполнением int32)
./app.out -m 17 17                 # кратно самому себе
# Проверка обработки ошибок (в задаче требуются ненулевые целые)
./app.out -m 0 5                   # ошибка: ноль в первом
./app.out -m 5 0                   # ошибка: ноль во втором

# --- Прямоугольный треугольник (-t epsilon a b c) ---
./app.out -t 1e-6 3 4 5            # классический прямоугольный
./app.out -t 1e-6 5 12 13          # другой Пифагоров тройка
./app.out -t 1e-6 1 1 1.41421356   # почти равнобедренный прямоугольный (с eps)
./app.out -t 1e-9 1 1 1.414213562  # более строгий eps — всё ещё прямоугольный
./app.out -t 1e-6 2 2 3            # не прямоугольный, но треугольник существует
./app.out -t 1e-6 1 2 3            # не образует треугольник (нарушение нерав-ва)
./app.out -t 1e-6 10 6 8           # прямоугольный в другой перестановке сторон
./app.out -t 1e-6 0 4 5            # ошибка: нулевая длина
./app.out -t 1e-6 -3 4 5           # ошибка: отрицательная длина
./app.out -t 1e-12 1e6 1e6 1414213.56237 # большие числа + строгий eps
